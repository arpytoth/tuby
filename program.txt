void fibo(int k)
{
    int elem;
    int pelem;
    int i;
    for (i = 1; i != k + 1; i++;)
    {
        if (i == 1 || i == 2)
        {  
            elem = 1;
            pelem = 1;
            print(1);
        }
        else
        {
            int newelem;
            newelem = elem + pelem;
            print(newelem);
            pelem = elem;
            elem = newelem;
        }
    }
}

fibo(10);
